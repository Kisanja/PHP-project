item: item_id(PK), item_type, item_description, price, product_img, weight
customer: username(PK), name, password, password_hint, email

total: tot_id(PK), username(FK of customer), item_id(FK of item), shipping_total, total, sub_total
customer_order: order_id(PK), day, date, customer_cart_id(FK of customer_cart), tot_id(FK of total)
